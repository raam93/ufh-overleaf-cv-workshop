% ------ REUSABLE COMMANDS ------

% Sidebar section heading: small-caps label + short rule underneath.
% \textsc{} renders text in Small Capitals — gives a refined, typeset feel
% without needing bold or large font sizes.
\newcommand{\sideheading}[1]{%
  \vspace{10pt}%
  {\small\textsc{\textbf{#1}}}\\[-2pt]%
  \rule{\linewidth}{0.3pt}%
  \vspace{4pt}%
}

% Main column section heading: larger small-caps + full-width rule.
\newcommand{\mainheading}[1]{%
  \vspace{10pt}%
  {\normalsize\textsc{\textbf{#1}}}\\[-4pt]%
  \rule{\linewidth}{0.4pt}%
  \vspace{4pt}%
}

% Job entry: bold title + italic dates on same line, company + location below.
% Args: {Title}{Company}{Location}{Dates}
\newcommand{\jobentry}[4]{%
  \noindent\textbf{#1} \hfill \textit{\small #4} \\%
  \textit{\small #2}, {\small #3}%
  \vspace{3pt}%
}
