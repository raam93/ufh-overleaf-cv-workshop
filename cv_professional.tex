% ------ REUSABLE SECTION COMMAND ------
% \rule{width}{height} draws a filled rectangle — used here as a divider line.
\newcommand{\prosection}[1]{%
  \vspace{10pt}%
  \noindent{\large \textsc{#1}}%   % \textsc = Small Capitals
  \vspace{3pt}\\%
  \noindent\rule{\textwidth}{0.4pt}%   % a thin full-width horizontal rule
  \vspace{4pt}%
}

% ------ ENTRY COMMAND ------
% Takes 4 arguments: {Job Title}{Company}{Location}{Dates}
\newcommand{\entry}[4]{%
  \noindent%
  \textbf{#1} \hfill \textit{#4} \\%     title + date
  \textit{#2} \hfill \small #3%          company + location
  \vspace{2pt}%
}
